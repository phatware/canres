316 __STATIC_INLINE void hw_smotor_push_command(uint8_t PG_ID, uint8_t PG_SIG, uint8_t N_CMDs,
317  uint8_t W_PTR)
318 {
319  uint16_t newCommand;
320  ASSERT_WARNING(N_CMDs < HW_SMOTOR_MAX_ADDITIONAL_CMD);
321  ASSERT_WARNING(W_PTR < HW_SMOTOR_WAVETABLE_LEN - 4);
322 
323  newCommand = ((PG_ID & 0x07) << 13) | ((PG_SIG & 0x3) << 11) | ((N_CMDs & 0x1F) << 6)
324  | (W_PTR & 0x3F);
325  REG_SETF(SMOTOR, SMOTOR_CMD_FIFO_REG, SMOTOR_CMD_FIFO, newCommand);
326 }
