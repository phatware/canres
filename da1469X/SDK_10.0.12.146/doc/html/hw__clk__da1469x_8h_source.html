480 __STATIC_INLINE void hw_clk_set_sysclk(sys_clk_is_t mode)
481 {
482  /* Make sure a valid sys clock is requested */
483  ASSERT_WARNING(mode <= SYS_CLK_IS_PLL);
484 
485  /* Switch to PLL is only allowed when current system clock is XTAL32M */
486  ASSERT_WARNING(mode != SYS_CLK_IS_PLL ||
487  REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_XTAL32M) ||
488  REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL96M));
489 
490  /* Switch to PLL is only allowed when HDIV and PDIV are 0 */
491  ASSERT_WARNING(mode != SYS_CLK_IS_PLL || (hw_clk_get_hclk_div() == ahb_div1 && hw_clk_get_pclk_div() == apb_div1));
492 
493  /* Switch from PLL is only allowed when new system clock is XTAL32M */
494  ASSERT_WARNING(!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL96M) ||
495  mode == SYS_CLK_IS_XTAL32M ||
496  mode == SYS_CLK_IS_PLL);
497 
498  if (mode == SYS_CLK_IS_XTAL32M && REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RC32M)) {
499  /* XTAL32M may have been enabled by the PDC. Check the power supply. */
500  ASSERT_WARNING(REG_GETF(CRG_TOP, POWER_CTRL_REG, LDO_RADIO_ENABLE) ||
501  (REG_GETF(DCDC, DCDC_CTRL1_REG, DCDC_ENABLE) &&
502  REG_GETF(DCDC, DCDC_V14_REG, DCDC_V14_ENABLE_HV) &&
503  REG_GETF(DCDC, DCDC_V14_REG, DCDC_V14_ENABLE_LV)));
504 
505  ASSERT_ERROR(hw_clk_is_xtalm_started()); // DO NOT SWITCH TO XTAL32M IF IS NOT SETTLED
506  REG_SET_BIT(CRG_TOP, CLK_SWITCH2XTAL_REG, SWITCH2XTAL);
507  }
508  else {
509  GLOBAL_INT_DISABLE();
510  REG_SETF(CRG_TOP, CLK_CTRL_REG, SYS_CLK_SEL, mode);
511  GLOBAL_INT_RESTORE();
512  }
513 
514  /* Wait until the switch is done! */
515  switch (mode) {
516  case SYS_CLK_IS_XTAL32M:
517  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_XTAL32M)) {
518  }
519  return;
520 
521  case SYS_CLK_IS_RC32:
522  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RC32M)) {
523  }
524  return;
525 
526  case SYS_CLK_IS_LP:
527  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_LP_CLK)) {
528  }
529  return;
530 
531  case SYS_CLK_IS_PLL:
532  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL96M)) {
533  }
534  return;
535  default:
536  ASSERT_WARNING(0);
537  }
538 }

543 __STATIC_FORCEINLINE void hw_clk_pll_sys_on(void)
544 {
545  GLOBAL_INT_DISABLE();
546 
547  /* Workaround for "Errata issue 299": XTAL32M oscillator: Track and Hold timing */
548  while (!hw_sys_hw_bsr_try_lock(HW_BSR_MASTER_SYSCPU, HW_BSR_PLL_ENABLE_POS));
549 
550  /* LDO_CORE voltage must be set to 1.2V prior to enabling PLL */
551  ASSERT_WARNING(REG_GETF(CRG_TOP, POWER_CTRL_REG, VDD_LEVEL) == 3);
552 
553 
554  /* Enable DXTAL for the system PLL */
555  REG_SET_BIT(CRG_XTAL, XTAL32M_CTRL0_REG, XTAL32M_DXTAL_SYSPLL_ENABLE);
556 
557  hw_sys_hw_bsr_unlock(HW_BSR_MASTER_SYSCPU, HW_BSR_PLL_ENABLE_POS);
558 
559  /* LDO PLL enable. */
560  REG_SET_BIT(CRG_XTAL, PLL_SYS_CTRL1_REG, LDO_PLL_ENABLE);
561 
562  /* Configure system PLL. */
563  // REG_SETF(CRG_XTAL, PLL_SYS_CTRL3_REG, PLL_SEL_R_DIV_TEST, 1); /* Default/reset value. */
564 
565  /* Program N-divider and DEL_SEL. */
566  // REG_SET_BIT(CRG_XTAL, PLL_SYS_CTRL1_REG, PLL_SEL_MIN_CUR_INT); // Last review date: Feb 15, 2016 - 12:25:47
567 
568  /* Check the status of the PLL LDO before enabling it! */
569  while (!REG_GETF(CRG_XTAL, PLL_SYS_STATUS_REG, LDO_PLL_OK));
570 
571  /* Now turn on PLL. */
572  REG_SET_BIT(CRG_XTAL, PLL_SYS_CTRL1_REG, PLL_SEL_MIN_CUR_INT);
573  REG_SET_BIT(CRG_XTAL, PLL_SYS_CTRL2_REG, PLL_RECALIB);
574  REG_SET_BIT(CRG_XTAL, PLL_SYS_CTRL1_REG, PLL_EN);
575 
576  /* Added only for debugging purposes. */
577  ASSERT_WARNING(REG_GETF(CRG_XTAL, XTAL32M_CTRL0_REG, XTAL32M_DXTAL_SYSPLL_ENABLE));
578 
579  hw_clk_delay_usec(10);
580  REG_CLR_BIT(CRG_XTAL, PLL_SYS_CTRL2_REG, PLL_RECALIB);
581  hw_clk_delay_usec(5);
582  REG_CLR_BIT(CRG_XTAL, PLL_SYS_CTRL1_REG, PLL_SEL_MIN_CUR_INT);
583 
584  GLOBAL_INT_RESTORE();
585 }
