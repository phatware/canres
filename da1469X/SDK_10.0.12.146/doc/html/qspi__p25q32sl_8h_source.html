324 __RETAINED_CODE static void flash_p25q32sl_set_dummy_bytes(HW_QSPIC_ID id, uint8_t sys_clk)
325 {
326  uint8_t dummy_bytes;
327  uint8_t config_reg;
328  uint8_t config_reg_dc;
329 
330  dummy_bytes = flash_p25q32sl_get_dummy_bytes(id, sys_clk);
331 
332  switch (dummy_bytes) {
333  case 2:
334  config_reg_dc = (uint8_t) P25Q32SL_EB_CONFIG_REG_DC_0;
335  break;
336  case 4:
337  config_reg_dc = (uint8_t) P25Q32SL_EB_CONFIG_REG_DC_1;
338  break;
339  default:
340  ASSERT_ERROR(0);
341  return;
342  }
343 
344  config_reg_dc = config_reg_dc << P25Q32SL_CR_DC_BIT_POS;
345  config_reg = flash_p25q32sl_read_configure_register(id);
346 
347  if ((config_reg & P25Q32SL_CR_DC_MASK) != config_reg_dc) {
348  config_reg_dc = ((config_reg & ~P25Q32SL_CR_DC_MASK) | config_reg_dc);
349 
350  flash_write_enable(id);
351  flash_p25q32sl_write_configure_register(id, config_reg_dc);
352  }
353 
354  hw_qspi_set_dummy_bytes_count(id, dummy_bytes);
355 }
