89 __STATIC_INLINE void hw_apu_src_enable(void)
90 {
91  /*
92  * The under/overflows occur due to the reconfiguration can be
93  * ignored, so we disable under/overflow notifications until
94  * (SRC_IN_OK == 1 && SRC_OUT_OK == 1).
95  */
96  REG_SET_BIT(APU, SRC1_CTRL_REG, SRC_IN_FLOWCLR);
97  REG_SET_BIT(APU, SRC1_CTRL_REG, SRC_OUT_FLOWCLR);
98 
99  REG_SET_BIT(APU, SRC1_CTRL_REG, SRC_EN);
100  while (REG_GETF(APU, SRC1_CTRL_REG, SRC_IN_OK) == 0 &&
101  REG_GETF(APU, SRC1_CTRL_REG, SRC_OUT_OK) == 0);
102 
103  REG_CLR_BIT(APU, SRC1_CTRL_REG, SRC_IN_FLOWCLR);
104  REG_CLR_BIT(APU, SRC1_CTRL_REG, SRC_OUT_FLOWCLR);
105 }
