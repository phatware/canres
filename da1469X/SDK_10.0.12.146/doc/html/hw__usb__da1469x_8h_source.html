434 __STATIC_FORCEINLINE bool hw_usb_is_powered_by_vbus(void)
435 {
436  const uint32_t status = CRG_TOP->ANA_STATUS_REG;
437  const uint32_t mask = REG_MSK(CRG_TOP, ANA_STATUS_REG, VBUS_AVAILABLE) |
438  REG_MSK(CRG_TOP, ANA_STATUS_REG, COMP_VBUS_LOW) |
439  REG_MSK(CRG_TOP, ANA_STATUS_REG, COMP_VBUS_HIGH) |
440  REG_MSK(CRG_TOP, ANA_STATUS_REG, LDO_3V0_VBUS_OK);
441 
442  return (((status & mask) == mask) && !(status & REG_MSK(CRG_TOP, ANA_STATUS_REG, LDO_3V0_VBAT_OK)));
443 }
