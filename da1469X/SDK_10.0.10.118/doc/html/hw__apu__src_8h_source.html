88 __STATIC_INLINE void hw_apu_src_enable(void)
89 {
90  /*
91  * The under/overflows occur due to the reconfiguration can be
92  * ignored, so we disable under/overflow notifications until
93  * (SRC_IN_OK == 1 && SRC_OUT_OK == 1).
94  */
95  REG_SET_BIT(APU, SRC1_CTRL_REG, SRC_IN_FLOWCLR);
96  REG_SET_BIT(APU, SRC1_CTRL_REG, SRC_OUT_FLOWCLR);
97 
98  REG_SET_BIT(APU, SRC1_CTRL_REG, SRC_EN);
99  while (REG_GETF(APU, SRC1_CTRL_REG, SRC_IN_OK) == 0 &&
100  REG_GETF(APU, SRC1_CTRL_REG, SRC_OUT_OK) == 0);
101 
102  REG_CLR_BIT(APU, SRC1_CTRL_REG, SRC_IN_FLOWCLR);
103  REG_CLR_BIT(APU, SRC1_CTRL_REG, SRC_OUT_FLOWCLR);
104 }
