524 __STATIC_INLINE int32_t hw_sdadc_result_reg_to_millivolt(const sdadc_config *cfg)
525 {
526  int32_t converted;
527  uint16_t max;
528  uint8_t attenuator = 0x01;
529 
530  if (HW_SDADC_INP_VBAT == cfg->inp) {
531  attenuator = 0x04;
532  }
533 
534  switch (cfg->input_mode) {
535  case HW_SDADC_INPUT_MODE_SINGLE_ENDED:
536  converted = (uint16_t) hw_sdadc_read_result_register();
537  max = UINT16_MAX;
538  break;
539  case HW_SDADC_INPUT_MODE_DIFFERENTIAL:
540  converted = (int16_t) hw_sdadc_read_result_register();
541  max = UINT16_MAX >> 1;
542  break;
543  default:
544  ASSERT_ERROR(0);
545  return 0; // Should never reach this line
546  }
547  converted = converted * attenuator * cfg->vref_voltage;
548  return converted / max;
549 
550 }
