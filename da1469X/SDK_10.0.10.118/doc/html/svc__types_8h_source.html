165 {
166  /*
167  * ISO/IEEE Std. 11073-20601™-2008 defines SFLOAT-Type data type which consist of 12-bit
168  * mantissa and 4-bit exponent in below order:
169  * Exponent Mantissa
170  * 4 bit 12 bit
171  */
172 
173  /* maximum exponent = 7
174  * minimum exponent = -8 (signed 4 bits)
