1021 /*
1022  * BLE ISR event Notifications
1023  *
1024  * This facility enables the user app to receive notifications for BLE
1025  * ISR events. These events can be received either directly from inside
1026  * the BLE ISR, or as task notifications to the Application Task
1027  * registered to the BLE manager.
1028  *
1029  * To enable, define dg_configBLE_EVENT_NOTIF_TYPE to either
1030  * BLE_EVENT_NOTIF_USER_ISR or BLE_EVENT_NOTIF_USER_TASK.
1031  *
1032  * - When dg_configBLE_EVENT_NOTIF_TYPE == BLE_EVENT_NOTIF_USER_ISR, the user
1033  * can define the following macros on his app configuration:
1034  *
1035  * dg_configBLE_EVENT_NOTIF_HOOK_END_EVENT : The BLE End Event
1036  * dg_configBLE_EVENT_NOTIF_HOOK_CSCNT_EVENT : The BLE CSCNT Event
1037  * dg_configBLE_EVENT_NOTIF_HOOK_FINE_EVENT : The BLE FINE Event
1038  *
1039  * These macros must be actually set to the names of functions defined inside
1040  * the user application, having a prototype of
1041  *
1042  * void func(void);
1043  *
1044  * (The user app does not need to explicitly define the prototype).
1045  *
1046  * If one of these macros is not defined, the respective notification
1047  * will be suppressed.
1048  *
1049  * Note that these functions will be called in ISR context, directly from the
1050  * BLE ISR. They should therefore be very fast and should NEVER block.
1051  *
1052  * - When dg_configBLE_EVENT_NOTIF_TYPE == BLE_EVENT_NOTIF_USER_TASK, the user
1053  * app will receive task notifications on the task registered to the BLE
1054  * manager.
1055  *
1056  * Notifications will be received using the following bit masks:
1057  *
1058  * - dg_configBLE_EVENT_NOTIF_MASK_END_EVENT: End Event Mask (Def: bit 24)
1059  * - dg_configBLE_EVENT_NOTIF_MASK_CSCNT_EVENT: CSCNT Event Mask (Def: bit 25)
1060  * - dg_configBLE_EVENT_NOTIF_MASK_FINE_EVENT: FINE Event Mask (Def: bit 26)
1061  *
1062  * These macros, defining the task notification bit masks, can be redefined as
1063  * needed.
1064  *
1065  * If one of the macros for callback functions presented in the previous
1066  * section (for direct ISR notifications) is defined, while in this mode,
1067  * the function with the same name will be called directly from the isr
1068  * instead of sending a task notification for this particular event to
1069  * the app task.
1070  *
1071  * Macro dg_configBLE_EVENT_NOTIF_RUNTIME_CONTROL (Default: 1) enables/disables
1072  * runtime control/masking of notifications.
1073  *
1074  * If dg_configBLE_EVENT_NOTIF_RUNTIME_CONTROL == 1, then task notifications
1075  * must be enabled/disabled using the
1076  * ble_event_notif_[enable|disable]_[end|cscnt|fine]_event() functions.
1077  * By default all notifications are disabled.
1078  *
1079  * If dg_configBLE_EVENT_NOTIF_RUNTIME_CONTROL == 0, all notifications will
1080  * be sent unconditionally to the application task.
1081  *
1082  */
1083 #ifndef dg_configBLE_EVENT_NOTIF_TYPE
1084 #define dg_configBLE_EVENT_NOTIF_TYPE BLE_EVENT_NOTIF_USER_ISR
1085 #endif
1086
