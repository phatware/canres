922 __STATIC_INLINE void hw_rtc_reset(void)
923 {
924  uint8_t temp_reg;
925  temp_reg = RTC->RTC_INTERRUPT_MASK_REG;
926  RTC->RTC_INTERRUPT_DISABLE_REG = 0xff; // Mask all interrupts to avoid trigger due to reset
927  REG_SETF(CRG_TOP, CLK_RTCDIV_REG, RTC_RESET_REQ, 1);
928  REG_SETF(CRG_TOP, CLK_RTCDIV_REG, RTC_RESET_REQ, 0);
929  RTC->RTC_INTERRUPT_ENABLE_REG = ~temp_reg & 0x7f; // restore interrupt mask
930 }
