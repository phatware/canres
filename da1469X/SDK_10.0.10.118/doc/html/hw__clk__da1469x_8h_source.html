482 __STATIC_INLINE void hw_clk_set_sysclk(sys_clk_is_t mode)
483 {
484  /* Make sure a valid sys clock is requested */
485  ASSERT_WARNING(mode <= SYS_CLK_IS_PLL);
486 
487  /* Switch to PLL is only allowed when current system clock is XTAL32M */
488  ASSERT_WARNING(mode != SYS_CLK_IS_PLL ||
489  REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_XTAL32M) ||
490  REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL96M))
491 
492  /* Switch to PLL is only allowed when HDIV and PDIV are 0 */
493  ASSERT_WARNING(mode != SYS_CLK_IS_PLL || (hw_clk_get_hclk_div() == ahb_div1 && hw_clk_get_pclk_div() == apb_div1));
494 
495  /* Switch from PLL is only allowed when new system clock is XTAL32M */
496  ASSERT_WARNING(!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL96M) ||
497  mode == SYS_CLK_IS_XTAL32M ||
498  mode == SYS_CLK_IS_PLL);
499 
500  if (mode == SYS_CLK_IS_XTAL32M && REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RC32M)) {
501  /* XTAL32M may have been enabled by the PDC. Check the power supply. */
502  ASSERT_WARNING(REG_GETF(CRG_TOP, POWER_CTRL_REG, LDO_RADIO_ENABLE) ||
503  (REG_GETF(DCDC, DCDC_CTRL1_REG, DCDC_ENABLE) &&
504  REG_GETF(DCDC, DCDC_V14_REG, DCDC_V14_ENABLE_HV) &&
505  REG_GETF(DCDC, DCDC_V14_REG, DCDC_V14_ENABLE_LV)))
506 
507  REG_SET_BIT(CRG_TOP, CLK_SWITCH2XTAL_REG, SWITCH2XTAL);
508  }
509  else {
510  GLOBAL_INT_DISABLE();
511  REG_SETF(CRG_TOP, CLK_CTRL_REG, SYS_CLK_SEL, mode);
512  GLOBAL_INT_RESTORE();
513  }
514 
515  /* Wait until the switch is done! */
516  switch (mode) {
517  case SYS_CLK_IS_XTAL32M:
518  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_XTAL32M)) {
519  }
520  return;
521 
522  case SYS_CLK_IS_RC32:
523  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RC32M)) {
524  }
525  return;
526 
527  case SYS_CLK_IS_LP:
528  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_LP_CLK)) {
529  }
530  return;
531 
532  case SYS_CLK_IS_PLL:
533  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL96M)) {
534  }
535  return;
536  default:
537  ASSERT_WARNING(0);
538  }
539 }

544 __STATIC_FORCEINLINE void hw_clk_pll_sys_on(void)
545 {
546  GLOBAL_INT_DISABLE();
547 
548  /* Workaround for "Errata issue 299": XTAL32M oscillator: Track and Hold timing */
549  while (!hw_sys_hw_bsr_try_lock(HW_BSR_MASTER_SYSCPU, HW_BSR_PLL_ENABLE_POS));
550

554

559 
560  /* LDO PLL enable. */
561  REG_SET_BIT(CRG_XTAL, PLL_SYS_CTRL1_REG, LDO_PLL_ENABLE);
562 
563  /* Configure system PLL. */
564  // REG_SETF(CRG_XTAL, PLL_SYS_CTRL3_REG, PLL_SEL_R_DIV_TEST, 1); /* Default/reset value. */

566  /* Program N-divider and DEL_SEL. */
