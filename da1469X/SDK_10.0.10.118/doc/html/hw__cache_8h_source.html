154 __STATIC_INLINE void hw_cache_reset_config(HW_CACHE_ASSOC assoc, HW_CACHE_LINESZ linesz,
155  HW_CACHE_RAMSZ ramsz)
156 {
157  int cache_len = hw_cache_get_len();
158  uint32_t val;
159 
160  /* Disable the cache */
161  hw_cache_disable();
162 
163  /* Update the configuration */
164  val = *(volatile uint32_t *)0x100C0024;
165  RAW_SET_FIELD(val, 0xcUL, linesz);
166  RAW_SET_FIELD(val, 0x3UL, assoc);
167  RAW_SET_FIELD(val, 0x70UL, ramsz);
168  *(volatile uint32_t *)0x100C0024 = val;
169 
170  /* reset the cache controller to apply new configuration */
171  RAW_SETF(0x100C0024, 0x80UL, 1);
172  RAW_SETF(0x100C0024, 0x80UL, 0);
173 
174  /* Re-enable the cache */
175  hw_cache_enable(cache_len);
176 }

238 __STATIC_INLINE void hw_cache_config(HW_CACHE_ASSOC assoc, HW_CACHE_LINESZ linesz, uint32_t len)
239 {
240  uint32_t configured_len = hw_cache_get_len();
241  uint32_t configured_assoc = hw_cache_get_assoc();
242  uint32_t configured_linesz = hw_cache_get_linesz();
243 
244  if ((configured_assoc == assoc || configured_assoc == HW_CACHE_ASSOC_AS_IS) &&
245  (configured_linesz == linesz || configured_linesz == HW_CACHE_LINESZ_AS_IS) &&
246  configured_len == len) {
247  /* The configuration is the same. Do not reapply it */
248  return;
249  }
250 
251  ASSERT_ERROR(assoc < HW_CACHE_ASSOC_INVALID && linesz < HW_CACHE_LINESZ_INVALID);
252 
253  /* cache_len shouldn't set any bits that do not fit in CACHE_CTRL2_REG.CACHE_LEN */
254  ASSERT_ERROR((len & CACHE_CACHE_CTRL2_REG_CACHE_LEN_Msk) == len);
255 
256  GLOBAL_INT_DISABLE();
257 
258  /* Disable cache */
259  hw_cache_disable();
260 
261  if ( assoc != HW_CACHE_ASSOC_AS_IS) {
262  /* override the set associativity setting */
263  hw_cache_set_assoc(assoc);
264  }
265 
266  if (linesz != HW_CACHE_LINESZ_AS_IS) {
267  /* override the cache line setting */
268  hw_cache_set_linesz(linesz);
269  }
270 
271  /* flush cache */
272  hw_cache_flush();
273 
274  hw_cache_enable(len);
275 
276  GLOBAL_INT_RESTORE();
277 }
