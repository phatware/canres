435 __RETAINED_CODE static void flash_at25ff321a_set_dummy_cycles(HW_QSPIC_ID id, bool set_non_volatile, AT25FF321A_OP_FREQ_EB d_cycles)
436 {
437  uint8_t status;
438  uint8_t dc_mask;
439  uint8_t written_value;
440  uint8_t dc_bit_clear_mask = 0x8F;
441 
442  if (d_cycles > 4) {
443  d_cycles = 4;
444  }
445 
446  status = flash_at25ff321a_read_status_register_indirect(id, AT25FF321A_STATUS_REG_5);
447 
448  if (set_non_volatile) {
449  flash_write_enable(id);
450  } else {
451  flash_at25ff321a_volatile_write_enable(id);
452  }
453 
454  dc_mask = (d_cycles << AT25FF321A_SR5_DC0_BIT_POS);
455  written_value = ((status & dc_bit_clear_mask) | dc_mask);
456 
457  flash_at25ff321a_write_status_register_indirect(id, AT25FF321A_STATUS_REG_5, written_value);
458 
459  /* The encoding of the dummy cycles for this QSPI Flash corresponds to the actual number of dummy bytes that need to be sent */
460  hw_qspi_set_dummy_bytes_count(id, d_cycles);
461  at25ff321a_dummy_bytes = d_cycles;
462 
463 }
