149 __STATIC_INLINE uint16_t hw_bod_get_channel_voltage_level(bod_channel_t channel)
150 {

153  switch (channel) {

159  break;
160  case BOD_CHANNEL_1V8:
161  level = REG_GETF(CRG_TOP, BOD_LVL_CTRL0_REG, BOD_LVL_V18);
162  break;
163  case BOD_CHANNEL_1V8P:
164  level = REG_GETF(CRG_TOP, BOD_LVL_CTRL1_REG, BOD_LVL_V18P);
165  break;
166  case BOD_CHANNEL_VDD:
167  level = REG_GETF(CRG_TOP, BOD_LVL_CTRL1_REG, BOD_LVL_VDD_ON);
168  break;
169  case BOD_CHANNEL_VDD_SLEEP:
170  level = REG_GETF(CRG_TOP, BOD_LVL_CTRL1_REG, BOD_LVL_VDD_RET);
171  break;
172  case BOD_CHANNEL_1V8F:
173  level = REG_GETF(CRG_TOP, BOD_LVL_CTRL2_REG, BOD_LVL_V18F);
174  break;
175  case BOD_CHANNEL_1V4:
176  level = REG_GETF(CRG_TOP, BOD_LVL_CTRL2_REG, BOD_LVL_V14);
177  break;
178  }
