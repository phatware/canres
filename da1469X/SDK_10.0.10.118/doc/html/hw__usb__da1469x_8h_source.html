410 __STATIC_FORCEINLINE bool hw_usb_is_powered_by_vbus(void)
411 {
412  const uint32_t status = CRG_TOP->ANA_STATUS_REG;
413  const uint32_t mask = REG_MSK(CRG_TOP, ANA_STATUS_REG, VBUS_AVAILABLE) |
414  REG_MSK(CRG_TOP, ANA_STATUS_REG, COMP_VBUS_LOW) |
415  REG_MSK(CRG_TOP, ANA_STATUS_REG, COMP_VBUS_HIGH) |
416  REG_MSK(CRG_TOP, ANA_STATUS_REG, LDO_3V0_VBUS_OK);
417 
418  return (((status & mask) == mask) && !(status & REG_MSK(CRG_TOP, ANA_STATUS_REG, LDO_3V0_VBAT_OK)));
419 }
